это ssh прокси jump
сначала ты указываешь ip и юхзера проксмоrc, потом порт целевой тачки (твоего миникуба) и его юзера с ip

ssh -J root@10.48.68.83 -p 13007 user@192.168.100.50

если делать это через .ssh/config выглядеть будет примерно так и никаких правил iptables не надо

Host proxmox
    HostName 10.48.68.83
    User root

Host vm-192-168-100-53
    HostName 192.168.100.53
    User user
    Port 11001
    ProxyJump proxmox
